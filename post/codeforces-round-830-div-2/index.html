A

div2最难的A题

容易发现操作代价不大于 333 ，因为我们可以对 ana_nan​ 和 an−1a_{n-1}an−1​ 分别操作满足要求。

只需要判断能否用 0/1/20/1/20/1/2 的代价满足要求即可。

Code

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 25;

int gcd(int a, int b) {
    return b ? gcd(b, a % b) : a;
}

int a[MAXN], b[MAXN], n, g, T;

int main() {
    ios::sync_with_stdio(false);
    cin >> T;
    while (T--) {
        cin >> n, g = 0;
        for (int i = 1; i <= n; i++) {
            cin >> a[i];
            g = gcd(g, a[i]);
        }
        if (g == 1) {
            cout << 0 << endl;
            continue;
        }
        int ans = 3;
        memcpy(b, a, sizeof(a));
        b[n] = gcd(n, b[n]);
        g = 0;
        for (int i = 1; i <= n; i++) g = gcd(g, b[i]);
        if (g == 1) {
            cout << 1 << endl;
            continue;
        }
        b[n] = a[n];
        b[n - 1] = gcd(b[n - 1], n - 1);
        g = 0;
        for (int i = 1; i <= n; i++) g = gcd(g, b[i]);
        if (g == 1) {
            cout << 2 << endl;
            continue;
        }
        cout << 3 << endl;
    }
}

B

从后往前扫一遍，对于连续 1 就跳过，否则找连续的 0 ，处理需要 222 次操作。

时间复杂度 O(∑n)O(\sum n)O(∑n) 。

Code

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5;

int T, n, ans, a[MAXN];

int main() {
    scanf("%d", &T);
    while (T--) {
        scanf("%d", &n);
        for (int i = 1; i <= n; i++)
            scanf("%1d", &a[i]);
        a[0] = -1;
        if (is_sorted(a + 1, a + n + 1)) {
            puts("0");
            continue;
        }
        int l = 1, r = n;
        while (a[l] == 0) l++;
        while (a[l] == 1) l++;
        ans = 0;
        if (a[r] == 0) {
            ans++;
            while (a[r] == 0) r--;
        }
        while (l < r) {
            while (a[r] == 1) r--;
            while (a[r] == 0) r--;
            ans += 2;
        }
        cout << ans << endl;
    }
}

C1

注意到异或相当于二进制下不进位加法，所以更大区间的答案一定更优。

对于只有一问的情况，答案就是 f(1,n)f(1,n)f(1,n) ，只需要求出最短的区间。

固定左端点并二分右端点所在位置即可，时间复杂度 O(nlog⁡n)O(n\log{n})O(nlogn) 。

Code

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5;

long long a[MAXN], sum[MAXN], sum_xor[MAXN];
int n, q, T, x, y, L, R;

inline bool check(int l, int mid) {
    long long f = (sum[mid] - sum[l - 1]) - (sum_xor[mid] ^ sum_xor[l - 1]);
    return f >= sum[n] - sum_xor[n];
}

int main() {
    scanf("%d", &T);
    while (T--) {
        scanf("%d%d", &n, &q);
        for (int i = 1; i <= n; i++) {
            scanf("%lld", &a[i]);
            sum[i] = sum[i - 1] + a[i];
            sum_xor[i] = sum_xor[i - 1] ^ a[i];
        }
        scanf("%d%d", &x, &y);
        L = 1, R = n;
        for (int i = 1; i <= n; i++) {
            int l = i, r = n;
            int res = -1;
            while (l <= r) {
                int mid = (l + r) >> 1;
                if (check(i, mid)) r = mid - 1, res = mid;
                else l = mid + 1;
            }
            if (res == -1) continue;
            if (res - i + 1 < R - L + 1) L = i, R = res;
        }
        printf("%d %d\n", L, R);
    }
}

C2

由于扩展区间所得到的答案一定更优，所以 [L,R][L,R][L,R] 区间的答案就是 f(L,R)f(L,R)f(L,R) ，我们仍然只需要最小化区间长度。

考虑从两边删数，删去的数需要满足 ∑x=⊕ x\sum x=\oplus\space x∑x=⊕ x 。

考虑异或本质是二进制下不进位加法，所以删去的数的每一位最多只有一个 111 。

由于 x<230x \lt 2^{30}x<230 ，所以最多只能从两边删去 303030 个数，枚举即可。

注意 000 的存在是不会对答案有影响的，但会影响枚举不超过30个这件事情，所以应当先删去 000 。

时间复杂度 O(n+qlog⁡n)O(n+q\log{n})O(n+qlogn) 。

Code

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5;

int n, q, T;
long long a[MAXN], b[MAXN], sum[MAXN], sum_xor[MAXN];

inline long long f(int l, int r) {
    long long s = sum[r] - sum[l - 1];
    long long s_xor = sum_xor[r] ^ sum_xor[l - 1];
    return s - s_xor;
}


int main() {
    ios::sync_with_stdio(false);
    cin >> T;
    while (T--) {
        cin >> n >> q;
        int tot = 0;
        for (int i = 1; i <= n; i++) {
            cin >> a[i];
            if (a[i]) b[++tot] = i;
        }
        n = tot;
        for (int i = 1; i <= n; i++) {
            sum_xor[i] = sum_xor[i - 1] ^ a[b[i]];
            sum[i] = sum[i - 1] + a[b[i]];
        }
        while (q--) {
            int l, r, k;
            cin >> l >> r, k = l;
            l = lower_bound(b + 1, b + tot + 1, l) - b;
            r = upper_bound(b + 1, b + tot + 1, r) - b - 1;
            if (l > r || n == 0) {
                cout << k << " " << k << endl;
                continue;
            }
            int L = l, R = r;
            for (int i = l; i <= min(r, l + 30); i++) {
                for (int j = r; j >= max(l, r - 30); j--) {
                    if (i <= j && f(i, j) == f(l, r) && b[j] - b[i] < b[R] - b[L])
                        L = i, R = j;
                }
            }
            cout << b[L] << " " << b[R] << endl;
        }
    }
}

}

#include <bits/stdc++.h>
using namespace std;

set<long long> s;
map<long long, set<long long>> maybe, factor;

char op[2];
long long x, q;

int main(){
    scanf("%lld", &q);
    while (q--) {
        scanf("%s%lld", op, &x);
        if (op[0] == '+') s.insert(x);
        if (op[0] == '-') {
            s.erase(x);
            for (auto it: factor[x]) maybe[it].insert(x);
        }
        if (op[0] == '?') {
            maybe[x].insert(x);
            factor[x].insert(x);
            auto it = maybe[x].begin();
            while (s.find(*it) != s.end()) {
                maybe[x].erase(it);
                if (maybe[x].empty()) maybe[x].insert(*it + x);
                it = maybe[x].begin();
                factor[*it].insert(x);
            }
            printf("%lld\n", *it);
        }
    }
}
