洛谷 P1462 题解

· 2021-11-24 · # 题解

题目链接

蒟蒻的第一篇蓝题题解

乍一看能够想到最短路，只需要跑一个从 111 到 nnn 的单源最短路即可，边权即为经过每条路需要扣掉的血量，点权即为需要交的钱。

题目让求经过的点交费最大一次的最小值，考虑二分答案，对于每一个点权，去跑一次最短路，看一看能不能回到 nnn 。

注意既然已经选出了最大的点权，就要考虑到跑最短路时其他经过的点的点权不能比这个点权要大，这是一个筛选的条件。

至于最短路算法，我采用的是堆优化 Dijkstra\text{Dijkstra}Dijkstra 。

具体细节看注释，这里不再赘述。

Code

#include <iostream>
#include <cstdio>
#include <algorithm>
#include <set>
#include <cstring>
#include <vector>
#include <utility>
#include <cstdlib>
using namespace std;

const int N = 1e4 + 5, M = 5e4 + 5;

typedef pair<int, int> PII;
const int INF = 0x3f3f3f3f;

template<class T> inline void read(T &x) {
    register int f = 1; x = 0; static char ch = getchar();
    while (ch < '0' || ch > '9') f = (ch == '-' ? -f : f), ch = getchar();
    while ('0' <= ch && ch <= '9') x = x * 10 + ch - '0', ch = getchar();
    x = (f == -1 ? -x : x);
}

struct Edge {
    int v;
    int next;
    int w;
    Edge() {}
    Edge(int _v, int _next, int _w): v(_v), next(_next), w(_w) {}
} e[M << 1]; // 无向边需要开两倍数组

int p[N << 1], d[N], c[N], t[N];
int n, m, b, eid;

inline void insert(int u, int v, int w) {
    e[eid].v = v;
    e[eid].next = p[u];
    e[eid].w = w;
    p[u] = eid++;
}

inline bool check(int x) {
    if (x < c[1] || x < c[n]) return false; 
    // 既然x是最大点权就不能比1号点和n号点点权还小，筛去
    set<PII> st;
    memset(d, 0x3f, sizeof(d));
    d[1] = 0;
    st.insert(make_pair(0, 1));
    while (st.size()) {
        int v = st.begin() -> second;
        st.erase(st.begin());
        for (int i = p[v]; i != -1; i = e[i].next) {
            if (d[e[i].v] > d[v] + e[i].w && c[e[i].v] <= x) { 
            	// 注意经过的点的点权要比x小
                st.erase(make_pair(d[e[i].v], e[i].v));
                d[e[i].v] = d[v] + e[i].w;
                st.insert(make_pair(d[e[i].v], e[i].v));
            }
        }
    }
    return d[n] <= b; // 是否能活着回到n，即最短路扣掉的血量是否小于上限
}

inline int binarysearch() {
    int l = 1, r = n;
    while (l < r) {
        int mid = (l + r) >> 1;
        if (check(t[mid])) r = mid;
        else l = mid + 1;
    }
    return t[l]; // 二分答案，对于每一个点权跑一次最短路
}

int main() {
    #ifndef ONLINE_JUDGE
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    #endif
    read(n), read(m), read(b);
    memset(p, -1, sizeof(p));
    for (int i = 1; i <= n; i++) read(c[i]);
    memcpy(t, c, sizeof(c));
    for (int i = 1; i <= m; i++) {
        int u, v, w;
        read(u), read(v), read(w);
        if (u == v) continue; // 去掉重边
        insert(u, v, w);
        insert(v, u, w);
    }
    sort(t + 1, t + n + 1);
    if (!check(t[n])) printf("AFK"); // 如果最大的点权都无法做到就无解
    else printf("%d", binarysearch());
    return 0;
}
