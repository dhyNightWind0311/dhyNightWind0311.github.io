0x00 前言

封闭的第一次模拟赛遇到了这个科技，因为完全不会喜提 20pts20 \text{pts}20pts 。

事后发现这个东西很好玩，赶紧学了一下。

0x01 概念

线段树合并是将两棵值域相同的权值线段树合并的算法，一般用来合并动态开点权值线段树。

普通权值线段树也可以合并，但是意义不大。

其本质是对应位置相加，虽然看起来很暴力，但一次合并的复杂度是优秀的 O(log⁡n)O(\log{n})O(logn) 。证明在此不展开。

实现上跟正常动态开点权值线段树几乎一致，差别只在合并上。

实现参考如下：

int merge(int u, int v, int l, int r) {
    if (u == 0) return v;
    if (v == 0) return u;
    if (l == r) {
        sum[u] += sum[v];
        return u;
    }
    int mid = (l + r) >> 1;
    ls[u] = merge(ls[u], ls[v], l, mid);
    rs[u] = merge(rs[u], rs[v], mid + 1, r);
    return pushup(u), u;
}

[ABC183F] Confluence

这是我找到的最模板的线段树合并

并查集维护连通性，在权值线段树上单点查询，注意并查集的合并方向要和权值线段树保持一致。

时间复杂度 O(nlog⁡n)O(n\log{n})O(nlogn) 。

Code

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5, MAXM = 64e5 + 5;

template<class T = int> inline T read() {
    T ret = 0;
    int f = 0, ch = getchar();
    while (ch < '0' || ch > '9') f ^= ch == '-', ch = getchar();
    while ('0' <= ch && ch <= '9') ret = ret * 10 + ch - '0', ch = getchar();
    return f ? -ret : ret;
}

int n, q, fa[MAXN], c[MAXN];

int get(int x) {
    return x == fa[x] ? x : fa[x] = get(fa[x]);
}

inline void un(int x, int y) {
    x = get(x), y = get(y);
    fa[y] = x;
}

int ls[MAXM], rs[MAXM], sum[MAXM], tot;

inline void pushup(int rt) {
    sum[rt] = sum[ls[rt]] + sum[rs[rt]];
}

void update(int &rt, int l, int r, int pos, int x) {
    if (!rt) rt = ++tot;
    if (l == r) {
        sum[rt] += x;
        return;
    }
    int mid = (l + r) >> 1;
    if (pos <= mid) update(ls[rt], l, mid, pos, x);
    else update(rs[rt], mid + 1, r, pos, x);
    pushup(rt);
}

int merge(int u, int v, int l, int r) {
    if (u == 0) return v;
    if (v == 0) return u;
    if (l == r) {
        sum[u] += sum[v];
        return u;
    }
    int mid = (l + r) >> 1;
    ls[u] = merge(ls[u], ls[v], l, mid);
    rs[u] = merge(rs[u], rs[v], mid + 1, r);
    return pushup(u), u;
}

int query(int rt, int l, int r, int pos) {
    if (!rt) return 0;
    if (l == r) return sum[rt];
    int mid = (l + r) >> 1;
    if (pos <= mid) return query(ls[rt], l, mid, pos);
    else return query(rs[rt], mid + 1, r, pos);
}

int rt[MAXN];

int main() {
    n = read(), q = read();
    for (int i = 1; i <= n; i++) {
        c[i] = read();
        update(rt[i], 1, n, c[i], 1);
        fa[i] = i;
    }
    while (q--) {
        int opt = read(), x = read(), y = read();
        if (opt == 1) {
            x = get(x), y = get(y);
            if (x == y) continue;
            un(x, y);
            merge(rt[x], rt[y], 1, n);
        } else {
            x = get(x);
            printf("%d\n", query(rt[x], 1, n, y));
        }
    }
}

P3224 [HNOI2012]永无乡

与上一题几乎一致，唯一需要改动的是权值线段树维护第 kkk 小。

其余不展开，时间复杂度 O(nlog⁡n)O(n\log{n})O(nlogn) 。

Code

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5, MAXM = MAXN << 6;

int n, m, q, fa[MAXN];

int get(int x) {
    return x == fa[x] ? x : fa[x] = get(fa[x]);
}

inline void un(int x, int y) {
    x = get(x), y = get(y);
    fa[y] = x;
}

int ls[MAXM], rs[MAXM], sum[MAXM], id[MAXM], tot;

inline void pushup(int rt) {
    sum[rt] = sum[ls[rt]] + sum[rs[rt]];
}

void update(int &rt, int l, int r, int pos, int x) {
    if (!rt) rt = ++tot;
    if (l == r) {
        sum[rt]++, id[rt] = x;
        return;
    }
    int mid = (l + r) >> 1;
    if (pos <= mid) update(ls[rt], l, mid, pos, x);
    else update(rs[rt], mid + 1, r, pos, x);
    pushup(rt);
}

int merge(int u, int v, int l, int r) {
    if (u == 0) return v;
    if (v == 0) return u;
    if (l == r) {
        sum[u] += sum[v], id[u] = id[v];
        return u;
    }
    int mid = (l + r) >> 1;
    ls[u] = merge(ls[u], ls[v], l, mid);
    rs[u] = merge(rs[u], rs[v], mid + 1, r);
    return pushup(u), u;
}

int query(int rt, int l, int r, int k) {
    if (k <= 0) return -1;
    if (l == r) return id[rt] ? id[rt] : -1;
    int mid = (l + r) >> 1;
    if (sum[ls[rt]] >= k) return query(ls[rt], l, mid, k);
    return query(rs[rt], mid + 1, r, k - sum[ls[rt]]);
}

int rt[MAXN];

template<class T = int> inline T read() {
    T ret = 0;
    int f = 0, ch = getchar();
    while (ch < '0' || ch > '9') f ^= ch == '-', ch = getchar();
    while ('0' <= ch && ch <= '9') ret = ret * 10 + ch - '0', ch = getchar();
    return f ? -ret : ret;
}

char opt[2];

int main() {
    n = read(), m = read();
    for (int i = 1, p; i <= n; i++) {
        fa[i] = i, p = read();
        update(rt[i], 1, n, p, i);
    }
    while (m--) {
        int u = get(read()), v = get(read());
        un(u, v);
        rt[u] = merge(rt[u], rt[v], 1, n);
    }
    q = read();
    while (q--) {
        scanf("%s", opt);
        int x = read(), y = read();
        if (opt[0] == 'Q') {
            x = get(x);
            printf("%d\n", query(rt[x], 1, n, y));
        } else {
            x = get(x), y = get(y);
            un(x, y);
            rt[x] = merge(rt[x], rt[y], 1, n);
        }
    }
}

0x03 应用

P4556 【模板】线段树合并

为什么拿这个题当模板啊

我们在树的每一个节点上开一棵权值线段树，通过树上差分把路径操作转化为单点操作。

于是问题转化为在权值线段树上单点修改，并记录出现次数最大值所在位置。

我们全部修改完后通过线段树合并的方式合并子树答案，时间复杂度 O(nlog⁡n)O(n\log{n})O(nlogn) 。

在这里采用了树剖的方式处理 lcalcalca ，比倍增要略优一些。

Code

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5, MAXM = 8e6 + 5, R = 1e5;

struct edge {
    int v, next;
    edge() = default;
} e[MAXN * 2];

int p[MAXN], n, m, eid;

inline void add(int u, int v) {
    e[eid].v = v;
    e[eid].next = p[u];
    p[u] = eid++;
}

int dep[MAXN], fa[MAXN], son[MAXN], top[MAXN], sz[MAXN];

void dfs(int u, int f) {
    sz[u] = 1;
    dep[u] = dep[f] + 1;
    fa[u] = f;
    son[u] = -1;
    for (int i = p[u]; i != -1; i = e[i].next) {
        int v = e[i].v;
        if (v == f) continue;
        dfs(v, u);
        sz[u] += sz[v];
        if (son[u] == -1 || sz[v] > sz[son[u]]) son[u] = v;
    }
}

void dfs_(int u, int t) {
    top[u] = t;
    if (son[u] == -1) return;
    dfs_(son[u], t);
    for (int i = p[u]; i != -1; i = e[i].next) {
        int v = e[i].v;
        if (v != fa[u] && v != son[u]) dfs_(v, v);
    }
}

int lca(int x, int y) {
    while (top[x] != top[y]) {
        if (dep[top[x]] < dep[top[y]]) swap(x, y);
        x = fa[top[x]];
    }
    if (dep[x] > dep[y]) swap(x, y);
    return x;
}

int pos[MAXM], sum[MAXM], ls[MAXM], rs[MAXM], tot;

inline void pushup(int rt) {
    if (sum[ls[rt]] >= sum[rs[rt]]) {
        sum[rt] = sum[ls[rt]];
        pos[rt] = pos[ls[rt]];
    } else {
        sum[rt] = sum[rs[rt]];
        pos[rt] = pos[rs[rt]];
    }
}

void update(int &rt, int l, int r, int x, int k) {
    if (!rt) rt = ++tot;
    if (l == r) {
        sum[rt] += k, pos[rt] = x;
        return;
    }
    int mid = (l + r) >> 1;
    if (x <= mid) update(ls[rt], l, mid, x, k);
    else update(rs[rt], mid + 1, r, x, k);
    pushup(rt);
}

int merge(int u, int v, int l, int r) {
    if (u == 0) return v;
    if (v == 0) return u;
    if (l == r) {
        sum[u] += sum[v];
        return u;
    }
    int mid = (l + r) >> 1;
    ls[u] = merge(ls[u], ls[v], l, mid);
    rs[u] = merge(rs[u], rs[v], mid + 1, r);
    pushup(u); return u;
}

int rt[MAXN], ans[MAXN];

void dfs(int u) {
    for (int i = p[u]; i != -1; i = e[i].next) {
        int v = e[i].v;
        if (v == fa[u]) continue;
        dfs(v);
        rt[u] = merge(rt[u], rt[v], 1, R);
    }
    ans[u] = pos[rt[u]];
    if (sum[rt[u]] == 0) ans[u] = 0;
}

int main() {
    ios::sync_with_stdio(false);
    cin >> n >> m;
    memset(p, -1, sizeof(p));
    for (int i = 1, u, v; i < n; i++) {
        cin >> u >> v;
        add(u, v);
        add(v, u);
    }
    dfs(1, 1), dfs_(1, 1);
    while (m--) {
        int x, y, z;
        cin >> x >> y >> z;
        update(rt[x], 1, R, z, 1);
        update(rt[y], 1, R, z, 1);
        int l = lca(x, y);
        update(rt[l], 1, R, z, -1);
        if (l != 1) update(rt[fa[l]], 1, R, z, -1);
    }
    dfs(1);
    for (int i = 1; i <= n; i++) cout << ans[i] << endl;
}

P1600 [NOIP2016 提高组] 天天爱跑步

考虑枚举观察员，计算每一个选手对观察员可能做出的贡献。

对于一条跑步路线 (s,t)(s,t)(s,t) ，设 l=lca⁡(s,t)l=\operatorname{lca}(s,t)l=lca(s,t) ，并把路径拆成 (s,l)(s,l)(s,l) 和 (l,t)(l,t)(l,t) 两段。

如果路径 (s,t)(s,t)(s,t) 能对观察员 xxx 做出贡献，则需要满足 dis(s,x)=wxdis(s,x)=w_xdis(s,x)=wx​ 。

分类讨论：

  • 如果 xxx 在路径 (s,l)(s,l)(s,l) 上，转换一下就是

deps=wx+depxdep_s=w_x+dep_x deps​=wx​+depx​

  • 如果 xxx 在路径 (l,t)(l,t)(l,t) 上，转换一下就是

deps+depx−2×depl=wxdep_s+dep_x-2\times dep_l=w_x deps​+depx​−2×depl​=wx​

2×depl−deps=depx−wx2\times dep_l-dep_s=dep_x-w_x 2×depl​−deps​=depx​−wx​

到这里我们要维护的东西应该很明确了。

在每一个点上开一棵权值线段树，通过树上差分把路径操作转换为单点操作。

对于路径 (s,t)(s,t)(s,t) ，在 sss 上标记 depsdep_sdeps​ 并在 lll 处撤销，在 ttt 上标记 2×depl−deps2\times dep_l-dep_s2×depl​−deps​ 并在 fa[l]fa[l]fa[l] 上撤销。

最后通过线段树合并统计答案。注意可能会产生负数，可以通过平移值域解决。

时间复杂度 O(nlog⁡n)O(n\log{n})O(nlogn) 。

#define fill(a, x) memset(a, x, sizeof(a))
using namespace std;

const int MAXN = 3e5 + 5;

struct edge {
    int v, next;
    edge() = default;
} e[MAXN * 2];

int p[MAXN], w[MAXN], n, m, eid;

inline void insert(int u, int v) {
    e[eid].v = v;
    e[eid].next = p[u];
    p[u] = eid++;
}

int dep[MAXN], fa[MAXN], son[MAXN], top[MAXN], sz[MAXN];

void dfs1(int u, int f) {
    sz[u] = 1;
    dep[u] = dep[f] + 1;
    fa[u] = f;
    son[u] = -1;
    for (int i = p[u]; i != -1; i = e[i].next) {
        int v = e[i].v;
        if (v == f) continue;
        dfs1(v, u);
        sz[u] += sz[v];
        if (son[u] == -1 || sz[v] > sz[son[u]]) son[u] = v;
    }
}

void dfs2(int u, int t) {
    top[u] = t;
    if (son[u] == -1) return;
    dfs2(son[u], t);
    for (int i = p[u]; i != -1; i = e[i].next) {
        int v = e[i].v;
        if (v != fa[u] && v != son[u]) dfs2(v, v);
    }
}

int lca(int x, int y) {
    while (top[x] != top[y]) {
        if (dep[top[x]] < dep[top[y]]) swap(x, y);
        x = fa[top[x]];
    }
    if (dep[x] > dep[y]) swap(x, y);
    return x;
}

    if (!rt) rt = ++tot;
    if (l == r) {
        sum[rt] += x;
        return;
    }
    int mid = (l + r) >> 1;
    if (pos <= mid) update(ls[rt], l, mid, pos, x);
    else update(rs[rt], mid + 1, r, pos, x);
}

int merge(int u, int v, int l, int r) {
    if (u == 0) return v;
    if (v == 0) return u;
    if (l == r) {
        sum[u] += sum[v];
        return u;
    }
    int mid = (l + r) >> 1;
    ls[u] = merge(ls[u], ls[v], l, mid);
    rs[u] = merge(rs[u], rs[v], mid + 1, r);
    return u;
}

    if (l == r) return sum[rt];
    int mid = (l + r) >> 1;

P3605 Promotion Counting

相比前两个题简单很多

先离散化，仍然是对每个点开权值线段树，答案只需要区间求和即可。

时间复杂度 O(nlog⁡n)O(n\log{n})O(nlogn) 。

Code

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5, MAXM = 128e5 + 5;

vector<int> g[MAXN];

int n, m, p[MAXN], tmp[MAXN];

int ls[MAXM], rs[MAXM], sum[MAXM], tot;

inline void pushup(int rt) {
    sum[rt] = sum[ls[rt]] + sum[rs[rt]];
}

void update(int &rt, int l, int r, int pos) {
    if (!rt) rt = ++tot;
    if (l == r) return sum[rt]++, void();
    int mid = (l + r) >> 1;
    if (pos <= mid) update(ls[rt], l, mid, pos);
    else update(rs[rt], mid + 1, r, pos);
    pushup(rt);
}

int merge(int u, int v, int l, int r) {
    if (u == 0) return v;
    if (v == 0) return u;
    if (l == r) {
        sum[u] += sum[v];
        return u;
    }
    int mid = (l + r) >> 1;
    ls[u] = merge(ls[u], ls[v], l, mid);
    rs[u] = merge(rs[u], rs[v], mid + 1, r);
    return pushup(u), u;
}

int query(int rt, int l, int r, int x, int y) {
    if (x <= l && r <= y) return sum[rt];
    if (y < l || r < x || rt == 0) return 0;
    int mid = (l + r) >> 1;
    return query(ls[rt], l, mid, x, y) + query(rs[rt], mid + 1, r, x, y);
}

int rt[MAXN], ans[MAXN];

void dfs(int u) {
    for (int v: g[u]) {
        dfs(v);
        rt[u] = merge(rt[u], rt[v], 1, m);
    }
    ans[u] = query(rt[u], 1, m, p[u] + 1, m);
}

int main() {
    ios::sync_with_stdio(false);
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> p[i];
        tmp[i] = p[i];
    }
    sort(tmp + 1, tmp + n + 1);
    m = unique(tmp + 1, tmp + n + 1) - tmp - 1;
    for (int i = 1; i <= n; i++) {
        p[i] = lower_bound(tmp + 1, tmp + m, p[i]) - tmp;
        update(rt[i], 1, m, p[i]);
    }
    for (int i = 2, fa; i <= n; i++) {
        cin >> fa;
        g[fa].push_back(i);
    }
    dfs(1);
    for (int i = 1; i <= n; i++) cout << ans[i] << endl;
}

P3521 Tree Rotations

一个线段树合并的非常规应用？

考虑交换一个点的左右子树，对于左右子树本身的逆序对数量是不会有影响的，有影响的只有横跨左右子树产生的逆序对。

我们在每个叶子节点上开一棵权值线段树，记录它本身的权值。

在不断向上合并的过程中，由于我们维护的是值域，所以很容易计算出横跨左右子树的逆序对数量。

我们分别对交换和不交换维护两个答案，最后取 min⁡\minmin 即可。

输入比较毒瘤。 时间复杂度 O(nlog⁡n)O(n\log{n})O(nlogn) 。

Code

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 * 64 + 5;

int n, ls[MAXN], rs[MAXN], sum[MAXN], tot;

long long ans, ans1, ans2;

inline void pushup(int rt) {
    sum[rt] = sum[ls[rt]] + sum[rs[rt]];
}

int update(int l, int r, int pos) {
    int u = ++tot;
    sum[u]++;
    if (l == r) return u;
    int mid = (l + r) >> 1;
    if (pos <= mid) ls[u] = update(l, mid, pos);
    else rs[u] = update(mid + 1, r, pos);
    return u;
}

int merge(int u, int v, int l, int r) {
    if (u == 0) return v;
    if (v == 0) return u;
    if (l == r) {
        sum[u] += sum[v];
        return u;
    }
    ans1 += 1ll * sum[ls[u]] * sum[rs[v]];  // swap
    ans2 += 1ll * sum[rs[u]] * sum[ls[v]];  // not swap
    int mid = (l + r) >> 1;
    ls[u] = merge(ls[u], ls[v], l, mid);
    rs[u] = merge(rs[u], rs[v], mid + 1, r);
    return pushup(u), u;
}

template<class T = int> inline T read() {
    T ret = 0;
    int f = 0, ch = getchar();
    while (ch < '0' || ch > '9') f ^= ch == '-', ch = getchar();
    while ('0' <= ch && ch <= '9') ret = ret * 10 + ch - '0', ch = getchar();
    return f ? -ret : ret;
}

int dfs() {
    int rt, val = read();
    if (val == 0) {
        int lc = dfs(), rc = dfs();
        ans1 = ans2 = 0;
        rt = merge(lc, rc, 1, n);
        ans += min(ans1, ans2);
    }
    else rt = update(1, n, val);
    return rt;
}

int main() {
    n = read();
    dfs();
    printf("%lld\n", ans);
}
