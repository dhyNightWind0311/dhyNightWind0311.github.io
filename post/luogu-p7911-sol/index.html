洛谷 P7911 题解

· 2021-11-13 · # 题解

题目链接

乍一看就是一道大模拟。

由于 ipipip 需要去重和编号对应的特性考虑使用 setsetset 和 mapmapmap 。

值得注意的一点是 ipipip 的判断细节很多，需要考虑到每一种情况，比如去除前导零，以及某个子段为空的情况。考场没判完全挂了25pts。

大模拟代码没什么好讲的，具体细节看注释。

Code

#include <iostream>
#include <cstdio>
#include <map>
#include <set>
#include <string>
using namespace std;
set<string> st;
map<string, int> mp;

inline bool cmp(string s1, string s2) { // 比较字符串大小
    int len1 = s1.size(), len2 = s2.size();
    if (len1 != len2) return len1 < len2;
    else return s1 <= s2;
}

inline bool iszero(string s1) {
    if (s1.size() == 1 && s1[0] == '0') return true; // 子段为“0”，符合题意
    if (s1.size() > 1 && s1[0] == '0') return false; // 子段有前导零，不符合题意
    if (s1.size() == 0) return false; // 子段为空，不符合题意
    return true;
}

inline bool check(string s) {
    int cnt1 = 0, cnt2 = 0;
    for (int i = 0; i < s.size(); i++) {
        if (s[i] == '.') cnt1++;
        if (s[i] == ':') cnt2++;
    } // 统计ip中“.”和“:”的个数
    if (cnt1 != 3 || cnt2 != 1) return false;
    int pos[4], cnt = 0;
    for (int i = 0; i < s.size(); i++) {
        if (s[i] == '.' || s[i] == ':') pos[cnt] = i, cnt++;
    } // 记录“.”和“:”的位置
    int p = s.find(':');
    if (p < pos[3]) return false; // 判断会不会有“.”和“:”颠倒位置的情况
    string l1 = s.substr(0, pos[0]);
    string l2 = s.substr(pos[0] + 1, pos[1] - pos[0] - 1);
    string l3 = s.substr(pos[1] + 1, pos[2] - pos[1] - 1);
    string l4 = s.substr(pos[2] + 1, pos[3] - pos[2] - 1);
    string l5 = s.substr(pos[3] + 1, s.size() - pos[3] - 1); // 暴力切割
    if (!iszero(l1) || !iszero(l2)  || !iszero(l3)  
     || !iszero(l4)  || !iszero(l5)) {
        return false;
    } // 去除有前导零或者子段为空的情况
    if (cmp(l1, "255") && cmp(l2, "255") && cmp(l3, "255") 
     && cmp(l4, "255") && cmp(l5, "65535")) return true; // 判断子段是否在要求范围内
    return false;
}

int main() {
    ios::sync_with_stdio(false);
    int n; cin >> n;
    for (int i = 1; i <= n; i++) {
        string s, ip;
        cin >> s >> ip;
        if (check(ip)) {
            if (s == "Server") {
                if (st.count(ip)) cout << "FAIL" << endl; // 已经有对应ip的服务器
                else {
                    cout << "OK" << endl;
                    st.insert(ip);
                    mp[ip] = i;
                } // 记录编号
            } else if (s == "Client") {
                if (st.count(ip)) cout << mp[ip] << endl; // 有对应ip的服务器
                else cout << "FAIL" << endl;
            }
        } else cout << "ERR" << endl;
    }
    return 0;
}
