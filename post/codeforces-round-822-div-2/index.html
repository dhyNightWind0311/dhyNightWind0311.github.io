Code

#include <bits/stdc++.h>

const int N = 305;

int main() {
    cin >> T;

        cin >> n;

        sort(a + 1, a + n + 1);

    }
}

C

感觉我的做法不是正经做法。

预处理 10610^6106 以内所有数的约数，对于每个需要删除的数，从它最小的约数开始枚举即可。

我们维护一个标记 mnimn_imni​ ，记录 iii 的倍数中没有被删除的最小值。利用这个标记很容易判断 jjj 能否被其约数 kkk 删除。同时，若 iii 不需要被删除，我们也需要更新 iii 的所有约数的 mnmnmn 标记。

时间复杂度 O(nlog⁡n+∑i=1ni×d(i))O(n\log{n}+\sum\limits_{i=1}^n i\times d(i))O(nlogn+i=1∑n​i×d(i)) 。？？？

注意到 10610^6106 以内正约数最多的数只有 240240240 个正约数，实际复杂度上界为 2.4×1082.4\times 10^82.4×108 左右，而且跑不满，在CF高效的测评机下可以卡过。

Code

#include <bits/stdc++.h>
using namespace std;

const int N = 1e6 + 5;

int t[N], n, q, mn[N];
char s[N];
vector<int> d[N];

inline void factor() {
    for (int i = 1; i < N; i++) {
        for (int j = i; j < N; j += i)
            d[j].emplace_back(i);
    }
}

int main() {
    scanf("%d", &q);
    factor();
    while (q--) {
        scanf("%d%s", &n, s + 1);
        for (int i = 1; i <= n; i++) {
            t[i] = s[i] - '0';
            mn[i] = n + 1;
        }
        long long ans = 0;
        for (int i = 1; i <= n; i++)
            for (int j: d[i]) {
                int flag = 0;
                if (t[i]) mn[j] = min(mn[j], i);
                if (mn[j] > n && !t[i]) ans += j, flag = 1;
                if (flag) break;
            }
        cout << ans << endl;
    }
}

D

赛时自以为写了正解，但是没交上，赛后发现假了。

考虑把整个数组按照 [1,k−1][1,k-1][1,k−1] 和 [k+1,n][k+1,n][k+1,n] 分为两段，对于每一段从后往前/从前往后划分为总和大于等于 000 的极小段，对于每一段，我们吸收这一段一定是不劣的，所以能吸收要尽量吸收。

考虑可以吸收一段的条件。我们记录段内的前/后缀和，容易发现吸收这一段必须满足当前生命值加上这一段内最小的前/后缀和大于等于 000 。即吸收 [l,r][l,r][l,r] 这一段前生命的最小值为

−min⁡i=lr∑j=liaj-\min\limits_{i=l}^r\sum\limits_{j=l}^i a_j −i=lminr​j=l∑i​aj​

这可以通过把前/后缀和插入到线段树中实现，时间复杂度 O(nlog⁡n)O(n\log{n})O(nlogn) 。

Code

#include <bits/stdc++.h>
#define ls (rt << 1)
#define rs (rt << 1 | 1)
using namespace std;

const int N = 2e5 + 5;

long long a[N], sum[N], t[N << 2], kth;
int n, k, T;
vector<pair<long long, long long>> L, R;

void pushup(int rt) {
    t[rt] = min(t[ls], t[rs]);
}

void build(int rt, int l, int r) {
    if (l == r) return t[rt] = 0, void();
    int mid = (l + r) >> 1;
    build(ls, l, mid);
    build(rs, mid + 1, r);
    pushup(rt);
}

void update(int rt, int l, int r, int pos, long long x) {
    if (l == r) return t[rt] = x, void();
    int mid = (l + r) >> 1;
    if (pos <= mid) update(ls, l, mid, pos, x);
    else update(rs, mid + 1, r, pos, x);
    pushup(rt);
}

long long query(int rt, int l, int r, int x, int y) {
    if (l > y || r < x) return 0;
    if (x <= l && r <= y) return t[rt];
    int mid = (l + r) >> 1;
    return min(query(ls, l, mid, x, y), query(rs, mid + 1, r, x, y));
}

void solve() {
    cin >> n >> k;
    sum[0] = 0;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        if (i == k) kth = a[i], a[i] = 0;
        sum[i] = sum[i - 1] + a[i];
    }
    L.clear(), build(1, 1, n);
    int l = k - 1;
    for (int i = k - 1; i >= 1; i--) {
        long long S = sum[l] - sum[i - 1];
        update(1, 1, n, i, S);
        if (S >= 0 || i == 1) {
            long long mn = query(1, 1, n, i, l);
            l = i - 1;
            L.emplace_back(-mn, S);
        }
    }
    R.clear(), build(1, 1, n);
    int r = k + 1;
    for (int i = k + 1; i <= n; i++) {
        long long S = sum[i] - sum[r - 1];
        update(1, 1, n, i, S);
        if (S >= 0 || i == n) {
            long long mn = query(1, 1, n, r, i);
            r = i + 1;
            R.emplace_back(-mn, S);
        }
    }
    reverse(L.begin(), L.end());
    reverse(R.begin(), R.end());
    while (true) {
        bool flag = false;
        while (!L.empty() && kth >= L.back().first) {
            flag = true;
            kth += L.back().second;
            L.pop_back();
        }
        while (!R.empty() && kth >= R.back().first) {
            flag = true;
            kth += R.back().second;
            R.pop_back();
        }
        if (L.empty() || R.empty()) {
            cout << "YES" << endl;
            return;
        }
        if (!flag) {
            cout << "NO" << endl;
            return;
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin >> T;
    while (T--) solve();
}
