using namespace std;

const int N = 1e5 + 5;

int x[N], t[N], n, T, mx, mn;

int main() {
    ios::sync_with_stdio(false);
    cin >> T;
    while (T--) {
        memset(x, 0, sizeof(x));
        memset(t, 0, sizeof(t));
        cin >> n;
        for (int i = 1; i <= n; i++) cin >> x[i];
        mx = -1e9, mn = 1e9;
        for (int i = 1; i <= n; i++) {
            cin >> t[i];
            mx = max(mx, max(x[i] + t[i], x[i] - t[i]));
            mn = min(mn, min(x[i] + t[i], x[i] - t[i]));
        }
        cout << (mx + mn) / 2;
        if ((mx + mn) & 1) cout << ".5";
        cout << endl;
    }
}

#include <bits/stdc++.h>
using namespace std;

const int N = 2e5 + 5;

int a[N], suf[N], cnt[N], n, T;
string s, ans;

int main() {
    ios::sync_with_stdio(false);
    cin >> T;
    while (T--) {
        memset(cnt, 0, sizeof(cnt));
        cin >> s;
        n = (int)s.size();
        for (int i = 0; i < n; i++) a[i] = s[i] - '0';
        suf[n] = 9;
        for (int i = n - 1; i >= 0; i--) suf[i] = min(suf[i + 1], a[i]);
        ans = "";
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < suf[i]; j++)
                while (cnt[j]) {
                    cnt[j]--;
                    ans += char(j + '0');
                }
            if (s[i] - '0' == suf[i]) ans += s[i];
            else cnt[min(a[i] + 1, 9)]++;
        }
        for (int i = 0; i < 10; i++)
            while (cnt[i]) {
                cnt[i]--;
                ans += char(i + '0');
            }
        cout << ans << endl;
    }
}

#include <bits/stdc++.h>
using namespace std;

const int N = 2e5 + 5;

string s1, s2;
int n, T, cnt, palin, a[N], b[N];

set<pair<int, int> > st;
map<pair<int, int>, int> mp;

int main() {
    ios::sync_with_stdio(false);
    cin >> T;
    while (T--) {
        cin >> n >> s1 >> s2;
        for (int i = 1; i <= n; i++) {
            a[i] = s1[i - 1] - 'a';
            b[i] = s2[i - 1] - 'a';
        }
        st.clear(), mp.clear();
        cnt = palin = 0;
        for (int i = 1; i <= n; i++) {
            auto p = make_pair(a[i], b[n - i + 1]);
            if (p.first > p.second) p = {p.second, p.first};
            if (st.count(p)) {
                mp[p]++;
                cnt += (mp[p] & 1) ? 1 : -1;
                if (p.first == p.second) palin += (mp[p] & 1) ? 1 : -1;
            } else {
                mp[p] = 1;
                st.insert(p);
                cnt++;
                if (p.first == p.second) palin++;
            }
        }
        if (cnt == palin && n % 2 == palin) cout << "YES" << endl;
        else cout << "NO" << endl;
    }
}
